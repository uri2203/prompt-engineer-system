// [Mantener variables globales iniciales] let currentImageB64 = ""; let currentAudioB64 = ""; async function ejecutarCCTV() { const visor = document.getElementById('cctv_visor'); const consola = document.getElementById('consola_salida'); const prompt = document.getElementById('input_prompt_visual').value; if(!prompt) return; visor.innerHTML = `
ENVIANDO ORDEN A NODO LOCAL...
`; try { const response = await fetch('/api/generate_image', { method: 'POST', headers: {'Content-Type': 'application/json'}, body: JSON.stringify({ prompt: prompt }) }); const data = await response.json(); if(data.status === "EN_COLA") { const tareaId = data.tarea_id; consola.innerHTML += `\n> [COLA] Orden #${tareaId} recibida por el Nodo. Renderizando con Máxima Calidad...`; // Iniciar Polling de resultado await esperarImagen(tareaId); } } catch (error) { visor.innerHTML = `[ ERROR DE ENLACE ]`; } } // NUEVA FUNCIÓN: Pregunta a Render cada 5 segundos si la imagen ya está lista async function esperarImagen(tareaId) { const visor = document.getElementById('cctv_visor'); const consola = document.getElementById('consola_salida'); const checkInterval = setInterval(async () => { try { const res = await fetch(`/api/check_image/${tareaId}`); const data = await res.json(); if(data.status === "READY") { clearInterval(checkInterval); currentImageB64 = data.image_url; visor.innerHTML = `
LIVE - NODO GAMMA
`; consola.innerHTML += `\n> [SISTEMA VISUAL] Imagen de alta fidelidad recibida.`; // Continuar con el siguiente paso del pipeline automáticamente if(currentAudioB64) await ejecutarEnsamblaje(); } } catch (e) { console.log("Esperando nodo..."); } }, 5000); }
